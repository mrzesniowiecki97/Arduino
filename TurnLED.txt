void setup()
{
  pinMode(8, OUTPUT);
  pinMode(7, INPUT_PULLUP);
}

void loop()
{
  int zmienna = digitalRead(8);
  digitalWrite(8, zmienna);
  if (digitalRead(7) == LOW && digitalRead(8) == LOW)
  {
    digitalWrite(8, HIGH);
  }
  else if (digitalRead(7) == LOW && digitalRead(8) == HIGH)
  {
    digitalWrite(8, LOW);
  }
}
